#include <stdio.h>

int main() {
    int n = 2;
    float matriz[2][3] = { {1, 2, 5}, {3, 4, 6} };
    float v;

    for (int i = 0; i < n; i++) {
    
        if (matriz[i][i] != 1) {
            v = matriz[i][i];
            for (int x = i; x < n + 1; x++) {
                matriz[i][x] = matriz[i][x] / v;
            }
        }

      
        for (int y = 0; y < n; y++) {
            if (y != i) {
                v = matriz[y][i];
                for (int x = i; x < n + 1; x++) {
                    matriz[y][x] = matriz[y][x] - v * matriz[i][x];
                }
            }
        }
    }

   
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < 3; j++) {
            printf("%f ", matriz[i][j]);
        }
        printf("\n");
    }

    return 0;
}
